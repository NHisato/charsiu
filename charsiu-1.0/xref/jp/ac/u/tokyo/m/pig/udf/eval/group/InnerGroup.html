123 		if (tJoinedValueBagSizesString[0] != "") {
124 			int tJoinedValueBagSizesLength = tJoinedValueBagSizesString.length;
125 			int[] tJoinedValueBagSizes = mJoinedValueBagSizes = new int[tJoinedValueBagSizesLength];
126 			for (int tIndex = tJoinedValueBagSizesLength - 1; tIndex >= 0; tIndex--) {

132 	private void selectModeOfTakeOutJoinedValue(String aModeOfTakeOutJoinedValueString, ModeOfTakeOutJoinedValue aDefaultMode) {
133 		if (aModeOfTakeOutJoinedValueString == null) {
134 			mModeOfTakeOutJoinedValue = aDefaultMode;
135 			return;
136 		}
137 		try {
138 			mModeOfTakeOutJoinedValue = ModeOfTakeOutJoinedValue.valueOf(aModeOfTakeOutJoinedValueString.toUpperCase());
139 		} catch (RuntimeException e) {
140 			LogUtil.errorIllegalModeName(mLog, ModeOfTakeOutJoinedValue.values(), aModeOfTakeOutJoinedValueString, e);
141 		}
142 	}

145 		ArrayList<String> tTargetJoinedValueBagNames = mTargetJoinedValueBagNames = new ArrayList<String>();
146 		if (aTargetJoinedValueBagNamesString == null) {

160 	@Override
161 	public Tuple exec(Tuple aInput) throws IOException {
162 		if (aInput == null)
163 			return null;
164 
165 		TupleFactory tTupleFactory = TupleFactory.getInstance();
166 		BagFactory tBagFactory = DefaultBagFactory.getInstance();
167 
168 		// processing target | 処理対象
169 		DataBag tTarget = DataType.toBag(aInput.get(0));
170 		DataBag tGroupColumn = DataType.toBag(aInput.get(1));
171 
172 		// InnerGroup container | InnerGroup の器
173 		GroupFilterFormat tGroupFilterFormat = mGroupFilterFormat;
174 		List<GroupFilterFormatGroup> tGroupList = tGroupFilterFormat.getGroupList();
175 		LinkedHashMap<GroupFilterFormatGroup, List<Tuple>> tInnerGroupMap = new LinkedHashMap<GroupFilterFormatGroup, List<Tuple>>();
176 		for (GroupFilterFormatGroup tCurrentGFFGroup : tGroupList) {
177 			tInnerGroupMap.put(tCurrentGFFGroup, new ArrayList<Tuple>());
178 		}
179 
180 		// Grouping
181 		Iterator<Tuple> tTargetIterator = tTarget.iterator();
182 		Iterator<Tuple> tGroupColumnIterator = tGroupColumn.iterator();
183 		int tOriginalColumnNum = mOriginalColumnNum;
184 		while (tGroupColumnIterator.hasNext()) {
185 			String tCurrentGroupColumnValue = (String) tGroupColumnIterator.next().get(0);
186 			Tuple tCurrentTargetTupleOrigin = tTargetIterator.next();
187 			// The contents of Bag influence it mutually for some reason when Pig executes eval UDF which appointed Bag in parallel.
188 			// As coping, I record the number of the columns from a schema of original Bag at the time of outputSchema, and only the number succeeds a column.
189 			// The interference of the value disappears in this at least UDF implementing. It is all right to execute InnerGroup in parallel.
190 			//
191 			// NOTE なぜだか Pig は Bag を指定した eval UDF を並列に動かすと、Bag の中身が相互に影響しあう。
192 			// 対処として、outputSchema の時点で元の Bag のスキーマからカラム数を記録し、その数だけカラムを引き継ぐ。
193 			// 少なくともこの実装をする UDF 同士では値の干渉はなくなる。InnerGroup を並列に実行するのは大丈夫。
194 			ArrayList<Object> tOriginBagColumns = new ArrayList<Object>();
195 			Iterator<Object> tColumnIterator = tCurrentTargetTupleOrigin.getAll().iterator();
196 			for (int tColumnCount = 0; tColumnCount < tOriginalColumnNum; tColumnCount++) {
197 				if (tColumnIterator.hasNext())
198 					tOriginBagColumns.add(tColumnIterator.next());
199 				else
200 					tOriginBagColumns.add(null);
201 			}
202 			int[] tJoinedValueBagIndexes = mJoinedValueBagIndexes;
203 			int[] tJoinedValueBagSizes = mJoinedValueBagSizes;
204 			// When target JoinedValueBag does not exist, overwrite tModeOfTakeOutJoinedValue in ALL_GROUP.
205 			// 対象とする JoinedValueBag が存在しない場合は tModeOfTakeOutJoinedValue を ALL_GROUP で上書きする。
206 			ModeOfTakeOutJoinedValue tModeOfTakeOutJoinedValue =
207 					tJoinedValueBagIndexes == null ? ModeOfTakeOutJoinedValue.ALL_GROUP : mModeOfTakeOutJoinedValue;
208 			for (GroupFilterFormatGroup tCurrentGFFGroup : tGroupList) {
209 				if (tCurrentGFFGroup.isMember(tCurrentGroupColumnValue)) {
210 					tInnerGroupMap.get(tCurrentGFFGroup).add(tTupleFactory.newTupleNoCopy(
211 							composeTransformedTuple(tCurrentGFFGroup, tModeOfTakeOutJoinedValue, tOriginBagColumns,
212 									tJoinedValueBagIndexes, tJoinedValueBagSizes)));
213 				}
214 			}
215 		}
216 
217 		// create return value | 戻り値作成
218 		ArrayList<Object> tResultTupleList = new ArrayList<Object>();
219 		for (GroupFilterFormatGroup tCurrentGFFGroup : tGroupList) {
220 			tResultTupleList.add(tCurrentGFFGroup.getGroupName());
221 			tResultTupleList.add(tBagFactory.newDefaultBag(tInnerGroupMap.get(tCurrentGFFGroup)));
222 		}
223 		return tTupleFactory.newTupleNoCopy(tResultTupleList);
224 	}

226 	private List<Object> composeTransformedTuple(GroupFilterFormatGroup aCurrentGFFGroup,
227 			ModeOfTakeOutJoinedValue aModeOfTakeOutJoinedValue, List<Object> aOriginBagColumns,
228 			int[] aJoinedValueBagIndexes, int[] aJoinedValueBagSizes) throws ExecException {
229 		switch (aModeOfTakeOutJoinedValue) {
230 		case OWN_GROUP:
231 			// 「 Bag{ Tuple( group_name : chararray, ... ) } 」
232 			// convert above structure into following structure | 上記構造を下記構造に変換
233 			// 「 ... 」
234 			// element 0 is unnecessary and adds element after 1 | 要素(0) は不要、要素(1) 以降を追加
235 			return composeTransformedTuple(aCurrentGFFGroup, aOriginBagColumns, aJoinedValueBagIndexes, aJoinedValueBagSizes, 1);
236 		case OWN_GROUP_WITH_GROUPNAME:
237 			// 「 Bag{ Tuple( group_name : chararray, ... ) } 」
238 			// convert above structure into following structure | 上記構造を下記構造に変換
239 			// 「 group_name : chararray, ... 」
240 			// adds element after 0 | 要素(0) 以降を追加
241 			return composeTransformedTuple(aCurrentGFFGroup, aOriginBagColumns, aJoinedValueBagIndexes, aJoinedValueBagSizes, 0);
242 		default:
243 			return aOriginBagColumns;
244 		}
245 	}

247 	private List<Object> composeTransformedTuple(GroupFilterFormatGroup aCurrentGFFGroup,
248 			List<Object> aOriginBagColumns, int[] aJoinedValueBagIndexes,
249 			int[] aJoinedValueBagSizes, int aOffsetValue) throws ExecException {
250 		List<Object> tResultTuple = new ArrayList<Object>();
251 		String tTargetGroupName = aCurrentGFFGroup.getGroupName();
252 		int tJoinedValueBagIndexesIndex = 0;
253 		int tOriginBagColumnsSize = aOriginBagColumns.size();
254 		int tJoinedValueBagIndexesIndexLimit = aJoinedValueBagIndexes.length - 1;
255 		for (int tIndex = 0; tIndex < tOriginBagColumnsSize; tIndex++) {
256 			if (tIndex == aJoinedValueBagIndexes[tJoinedValueBagIndexesIndex]) {
257 				// if JoinedValueBag position | JoinedValueBag の位置なら
258 				DataBag tCurrentTargetJoinedValueBag = (DataBag) aOriginBagColumns.get(tIndex);
259 				Iterator<Tuple> tCurrentTargetJoinedValueBagIterator = tCurrentTargetJoinedValueBag.iterator();
260 				Tuple tCurrentGroupJoinedValueTuple = null;
261 				while (tCurrentTargetJoinedValueBagIterator.hasNext()) {
262 					Tuple tCurrentTargetJoinedValueTuple = tCurrentTargetJoinedValueBagIterator.next();
263 					String tCurrentTargetJoinedValueTupleGroupName = (String) tCurrentTargetJoinedValueTuple.get(0);
264 					if (tTargetGroupName.equals(tCurrentTargetJoinedValueTupleGroupName)) {
265 						tCurrentGroupJoinedValueTuple = tCurrentTargetJoinedValueTuple;
266 						break;
267 					} else
268 						continue;
269 				}
270 				if (tCurrentGroupJoinedValueTuple == null) {
271 					// When Tuple belonging to this tTargetGroupName does not exist, set null
272 					// この tTargetGroupName に所属する Tuple が存在しない場合、 null 埋め
273 					int tCurrentBagSize = aJoinedValueBagSizes[tJoinedValueBagIndexesIndex];
274 					for (int tTupleIndex = 0; tTupleIndex < tCurrentBagSize; tTupleIndex++) {
275 						tResultTuple.add(null);
276 					}
277 				} else {
278 					List<Object> tTupleElementList = tCurrentGroupJoinedValueTuple.getAll();
279 					int tCurrentBagSize = aJoinedValueBagSizes[tJoinedValueBagIndexesIndex] + aOffsetValue;
280 					int tTupleElementListSize = tTupleElementList.size();
281 					for (int tTupleElementIndex = aOffsetValue; tTupleElementIndex < tCurrentBagSize; tTupleElementIndex++) {
282 						if (tTupleElementIndex < tTupleElementListSize) {
283 							tResultTuple.add(tTupleElementList.get(tTupleElementIndex));
284 						} else {
285 							tResultTuple.add(null);
286 						}
287 					}
288 				}
289 				if (tJoinedValueBagIndexesIndex < tJoinedValueBagIndexesIndexLimit)
290 					tJoinedValueBagIndexesIndex++;
291 			} else {
292 				tResultTuple.add(aOriginBagColumns.get(tIndex));
293 			}
294 		}
295 		return tResultTuple;
296 	}

312 	@Override
313 	public Schema outputSchema(Schema aInput) {
314 		Schema tResultSchema = new Schema();
315 
316 		List<FieldSchema> tInputFields = aInput.getFields();
317 		FieldSchema tInputTarget = tInputFields.get(0);
318 		Schema tInputTargetSchema = tInputTarget.schema;
319 		try {
320 			// record necessary schema information at processing | 処理時に必要なスキーマ情報を記録する
321 			Schema tTransformedTargetTupleSchema = commitSchemaInformation(tInputTargetSchema);
322 
323 			Schema tInnerSchema = new Schema();
324 
325 			String tAliasGroupNamePrefix = AliasConstants.INNER_GROUP_OUT_ALIAS_GROUP_NAME_PREFIX;
326 			String tAliasGroupBagPrefix = AliasConstants.INNER_GROUP_OUT_ALIAS_GROUP_BAG_PREFIX;
327 			byte tDataTypeChararray = DataType.CHARARRAY;
328 			byte tDataTypeBag = DataType.BAG;
329 
330 			List<GroupFilterFormatGroup> tGroupList = mGroupFilterFormat.getGroupList();
331 			for (GroupFilterFormatGroup tCurrentGFFGroup : tGroupList) {
332 				// Group Name | グループ名
333 				tInnerSchema.add(new FieldSchema(tAliasGroupNamePrefix + tCurrentGFFGroup.getGroupName()
334 						, tDataTypeChararray));
335 				// Structure of Bag of each group that is structure of input Bag | 各グループの Bag の構造、つまり入力 Bag の構造
336 				tInnerSchema.add(new FieldSchema(
337 						tAliasGroupBagPrefix + tCurrentGFFGroup.getGroupName()
338 						, tTransformedTargetTupleSchema, tDataTypeBag));
339 			}
340 
341 			tResultSchema.add(new FieldSchema(AliasConstants.INNER_GROUP_OUT_ALIAS_TOP
342 					, tInnerSchema, DataType.TUPLE));
343 		} catch (FrontendException e) {
344 			throw new RuntimeException(e);
345 		}
346 
347 		return tResultSchema;
348 	}

350 	/**
351 	 * Construction of output Schema depending on Mode <br>
352 	 * Mode に応じた 出力Schema の構築 <br>
353 	 */
354 	private Schema commitSchemaInformation(Schema aInputTargetSchema) throws FrontendException {
355 		// record the number of input Bag columns for exec | exec のために元Bagのカラム数を記録
356 		Properties tProperties = UDFContext.getUDFContext().getUDFProperties(this.getClass());
357 		// Possibly this may become needless after pig-0.9.
358 		// XXX pig-0.8系 から 0.9系 間の仕様変更に対応。もしかしたらこれが不要になってるかも知れない。
359 		Schema tInputTargetTupleSchema = aInputTargetSchema.getFields().get(0).schema;
360 		tProperties.setProperty(mKeyOriginalColumnNum, String.valueOf(tInputTargetTupleSchema.size()));
361 		// record columns num and position of input Bag "<joined_value> : Bag{ Tuple( group_name : chararray ) }" for exec
362 		// exec のために元Bagの <joined_value> : Bag{ Tuple( group_name : chararray ) } の 位置・カラム数 を記録。
363 		ModeOfTakeOutJoinedValue tModeOfTakeOutJoinedValue = mModeOfTakeOutJoinedValue;
364 		if (tModeOfTakeOutJoinedValue == ModeOfTakeOutJoinedValue.ALL_GROUP)
365 			return aInputTargetSchema;
366 		StringBuilder tKeyJoinedValueBagIndexesStringBuilder = new StringBuilder();
367 		StringBuilder tKeyJoinedValueBagSizesStringBuilder = new StringBuilder();
368 		List<FieldSchema> tInputTargetTupleFields = tInputTargetTupleSchema.getFields();
369 		int tInputTargetTupleFieldsSize = tInputTargetTupleFields.size();
370 		ArrayList<String> tTargetJoinedValueBagNames = mTargetJoinedValueBagNames;
371 		String tValueJoinOutAliasInnerGroupName = AliasConstants.VALUE_JOIN_OUT_ALIAS_INNER_GROUP_NAME;
372 		// rebuild a schema of aInputTargetSchema depending on Mode
373 		// aInputTargetSchema のスキーマを Mode に応じて再構築
374 		Schema tTransformedTargetTupleSchema = new Schema();
375 		for (int tIndex = 0; tIndex < tInputTargetTupleFieldsSize; tIndex++) {
376 			FieldSchema tCurrentInputTargetTupleFieldSchema = tInputTargetTupleFields.get(tIndex);
377 			if (StringUtil.equalsWord(tCurrentInputTargetTupleFieldSchema.alias, tTargetJoinedValueBagNames)) {
378 				// acquire alias of Field hoping that it is group_name of the "Bag{ Tuple( gourp_name : chararray ) }"
379 				// Bag{ Tuple( gourp_name : chararray ) } の group_name であることを期待するフィールドの alias を取得
380 				String tFirstFieldName = tCurrentInputTargetTupleFieldSchema.schema.getFields().get(0).schema.getFields().get(0).alias;
381 				if (tFirstFieldName.equals(tValueJoinOutAliasInnerGroupName)) {
382 					tKeyJoinedValueBagIndexesStringBuilder.append(tIndex);
383 					tKeyJoinedValueBagIndexesStringBuilder.append(',');
384 					tKeyJoinedValueBagSizesStringBuilder.append(
385 							composeTransformedSchema(tModeOfTakeOutJoinedValue, tTransformedTargetTupleSchema, tCurrentInputTargetTupleFieldSchema));
386 					tKeyJoinedValueBagSizesStringBuilder.append(',');
387 				} else {
388 					tTransformedTargetTupleSchema.add(tCurrentInputTargetTupleFieldSchema);
389 				}
390 			} else {
391 				tTransformedTargetTupleSchema.add(tCurrentInputTargetTupleFieldSchema);
392 			}
393 		}
394 		if (tKeyJoinedValueBagIndexesStringBuilder.length() > 0) {
395 			tProperties.setProperty(mKeyJoinedValueBagIndexes,
396 					tKeyJoinedValueBagIndexesStringBuilder.substring(0, tKeyJoinedValueBagIndexesStringBuilder.length() - 1));
397 			tProperties.setProperty(mKeyJoinedValueBagSizes,
398 					tKeyJoinedValueBagSizesStringBuilder.substring(0, tKeyJoinedValueBagSizesStringBuilder.length() - 1));
399 		}
400 		return tTransformedTargetTupleSchema;
401 	}

403 	/**
404 	 * @return
405 	 *         Number of element of aInputTargetTupleFieldSchema after the conversion <br>
406 	 *         変換後の aInputTargetTupleFieldSchema の要素数 <br>
407 	 */
408 	private int composeTransformedSchema(ModeOfTakeOutJoinedValue aModeOfTakeOutJoinedValue, Schema aTransformedTargetTupleSchema, FieldSchema aInputTargetTupleFieldSchema) {
409 		switch (aModeOfTakeOutJoinedValue) {
410 		case OWN_GROUP:
411 			// 「 Bag{ Tuple( group_name : chararray, ... ) } 」
412 			// convert above structure into following structure | 上記構造を下記構造に変換
413 			// 「 ... 」
414 			// element 0 is unnecessary and adds element after 1 | 要素(0) は不要、要素(1) 以降を追加
415 			return appendSchemas(aTransformedTargetTupleSchema, aInputTargetTupleFieldSchema, 1);
416 		case OWN_GROUP_WITH_GROUPNAME:
417 			// 「 Bag{ Tuple( group_name : chararray, ... ) } 」
418 			// convert above structure into following structure | 上記構造を下記構造に変換
419 			// 「 group_name : chararray, ... 」
420 			// adds element after 0 | 要素(0) 以降を追加
421 			return appendSchemas(aTransformedTargetTupleSchema, aInputTargetTupleFieldSchema, 0);
422 		default:
423 			return 0;
424 		}
425 	}

427 	private int appendSchemas(Schema aTransformedTargetTupleSchema, FieldSchema aInputTargetTupleFieldSchema, int aOffsetValue) {
428 		List<FieldSchema> tList = aInputTargetTupleFieldSchema.schema.getFields().get(0).schema.getFields();
429 		int tSize = tList.size();
430 		for (int tIndex = aOffsetValue; tIndex < tSize; tIndex++) {
431 			aTransformedTargetTupleSchema.add(tList.get(tIndex));
432 		}
433 		return tSize - aOffsetValue;
434 	}
