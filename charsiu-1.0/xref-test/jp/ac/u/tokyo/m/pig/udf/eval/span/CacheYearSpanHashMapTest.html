54  	public void testTimeSingleYearMapFactory() throws ParseException {
55  		CacheYearSpanHashMap tYearSpanMap = CacheYearSpanHashMap.INSTANCE;
56  		String tBaseDateString = "20110401";
57  
58  		// キャッシュするので２回目以降は高速
59  		checkGetYearSpan(tYearSpanMap, tBaseDateString);
60  		checkGetYearSpan(tYearSpanMap, tBaseDateString);
61  	}
62  
63  	private void checkGetYearSpan(CacheYearSpanHashMap aYearSpanMap, String aBaseDateString) throws ParseException {
64  		SimpleDateFormat tDateFormat = StringFormatConstants.FORMAT_DATE.get();
65  		Calendar tBaseDate = Calendar.getInstance();
66  		tBaseDate.setTime(tDateFormat.parse(aBaseDateString));
67  		long tStartTime = System.currentTimeMillis();
68  		for (Long tCount = 0L; tCount < 10000; tCount++) {
69  			aYearSpanMap.getYearSpan(aBaseDateString, tDateFormat.format(tBaseDate.getTime()));
70  			tBaseDate.add(Calendar.DATE, 1);
71  		}
72  		System.out.println((System.currentTimeMillis() - tStartTime) + " ms");
73  	}
