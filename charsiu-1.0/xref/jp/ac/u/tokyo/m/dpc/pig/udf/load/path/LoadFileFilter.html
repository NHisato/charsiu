31  /**
32   * This class filters files from specific file name pattern and SUBMIT_NUMBER. <br>
33   * <br>
34   * 特定のファイル名パターンと SUBMIT_NUMBER から、読み込むファイルをフィルタします。 <br>
35   */
36  public class LoadFileFilter {
37  
38  	// -----------------------------------------------------------------------------------------------------------------
39  
40  	private LinkedHashMap<DirectoryNameElements, Pattern> mFileNamePatterns;
41  
42  	// -----------------------------------------------------------------------------------------------------------------
43  
44  	public LoadFileFilter(List<String> aFilePatternStrings, LinkedHashMap<String, String> aReplaceWords) {
45  		LinkedHashMap<String, String> tReplaceWordsQuoted = new LinkedHashMap<String, String>();
46  		for (Entry<String, String> tEntry : aReplaceWords.entrySet()) {
47  			tReplaceWordsQuoted.put(LoadTermUtil.quoteReplaceWord(tEntry.getKey()),
48  					StringUtil.quoteRegularExpressionGroup(tEntry.getValue()));
49  		}
50  		LinkedHashMap<DirectoryNameElements, Pattern> tFileNamePatterns = mFileNamePatterns = new LinkedHashMap<DirectoryNameElements, Pattern>();
51  		int tSize = aFilePatternStrings.size();
52  		for (int tIndex = 0; tIndex < tSize;) {
53  			DirectoryNameElements tDirectoryNameElements = new DirectoryNameElements(aFilePatternStrings.get(tIndex++), aReplaceWords, tReplaceWordsQuoted);
54  			tFileNamePatterns.put(tDirectoryNameElements, Pattern.compile(tDirectoryNameElements.getCurrentContentName()));
55  		}
56  	}
57  
58  	// -----------------------------------------------------------------------------------------------------------------
59  
60  	public void filterSubmitNumber(List<FileStatus> aFileStatusList,
61  			Map<String, FileStatusWithVersion> aResultContainsFixNumberPath,
62  			List<FileStatus> aResultPath) {
63  		for (FileStatus tCurrentFileStatus : aFileStatusList) {
64  			filterSubmitNumber(tCurrentFileStatus, aResultContainsFixNumberPath, aResultPath);
65  		}
66  	}
67  
68  	private void filterSubmitNumber(FileStatus aFileStatus,
69  			Map<String, FileStatusWithVersion> aResultContainsFixNumberPath,
70  			List<FileStatus> aResultPath) {
71  		String tName = aFileStatus.getPath().getName();
72  		for (Entry<DirectoryNameElements, Pattern> tCurrentFileNamePattern : mFileNamePatterns.entrySet()) {
73  			Matcher tMatcher = tCurrentFileNamePattern.getValue().matcher(tName);
74  			if (tMatcher.matches()) {
75  				DirectoryNameElements tCurrentDirectoryContents = tCurrentFileNamePattern.getKey();
76  				if (tCurrentDirectoryContents.isReplaced(PathConstants.SUBMIT_NUMBER)) {
77  					// HOSPITAL_CODE, yyyyMM, SUBMIT_NUMBER are reserved word.
78  					// HOSPITAL_CODE, yyyyMM, SUBMIT_NUMBER を扱えるように。
79  					LinkedHashMap<String, String> tIdentities = new LinkedHashMap<String, String>();
80  					int tIndex = 1;
81  					for (String tReplaceWord : tCurrentDirectoryContents.getReplaceWordsSequence()) {
82  						tIdentities.put(tReplaceWord, tMatcher.group(tIndex++));
83  					}
84  					// register aFileStatus of newest SUBMIT_NUMBER (key = "HOSPITAL_CODE + yyyyMM")
85  					// HOSPITAL_CODE + yyyyMM を鍵として、最も新しい SUBMIT_NUMBER の aFileStatus を登録
86  					String tStatusKey = tIdentities.get(PathConstants.HOSPITAL_CODE) + tIdentities.get(PathConstants.yyyyMM);
87  					// 既に tStatusKey が有れば、バージョン比較して追加
88  					FileStatusWithVersion tFileStatusVersion = aResultContainsFixNumberPath.get(tStatusKey);
89  					if (tFileStatusVersion == null) {
90  						aResultContainsFixNumberPath.put(tStatusKey, new FileStatusWithVersion(aFileStatus, tIdentities.get(PathConstants.SUBMIT_NUMBER)));
91  					} else {
92  						if (tFileStatusVersion.getVersion().compareTo(tIdentities.get(PathConstants.SUBMIT_NUMBER)) < 0) {
93  							aResultContainsFixNumberPath.put(tStatusKey, new FileStatusWithVersion(aFileStatus, tIdentities.get(PathConstants.SUBMIT_NUMBER)));
94  						}
95  					}
96  				} else
97  					aResultPath.add(aFileStatus);
98  			} else
99  				continue;
100 		}
101 	}
